98      private static Map<String, String> createNameToVersionMap() {
99  
100         List<NameAndVersion> nameAndVersions = new ArrayList<>(oktaComponentsFromVersionMetadata());
101         nameAndVersions.add(getShiroInfo());                // shiro
102         nameAndVersions.add(getSpringFrameworkInfo());      // spring
103         nameAndVersions.add(getSpringBootInfo());           // spring-boot
104         nameAndVersions.add(getSpringSecurityInfo());       // spring-security
105         nameAndVersions.add(getWebServerInfo());            // tomcat | jetty | jboss | websphere | glassfish | weblogic | wildfly
106         nameAndVersions.add(getJavaSDKRuntimeInfo());       // java
107         nameAndVersions.add(getOSInfo());                   // Mac OS X
108         nameAndVersions.add(getAndroidInfo());              // android
109         return nameAndVersions.stream()
110                 .filter(Objects::nonNull)
111                 .collect(LinkedHashMap::new, // keep order
112                         (map, info) -> map.put(info.name, info.version),
113                         Map::putAll);
114     }

141     private static NameAndVersion getWebServerInfo() {
142         NameAndVersion webServerInfo;
143         //Glassfish uses Tomcat internally, therefore the Glassfish check must be carried out before Tomcat's
144         webServerInfo = getFullEntryStringUsingManifest(WEB_SERVER_GLASSFISH_CLASS, WEB_SERVER_GLASSFISH_ID);
145         if(webServerInfo != null) {
146             return webServerInfo;
147         }
148         webServerInfo = getFullEntryStringUsingManifest(WEB_SERVER_TOMCAT_BOOTSTRAP_CLASS, WEB_SERVER_TOMCAT_ID);
149         if(webServerInfo != null) {
150             return webServerInfo;
151         }
152         webServerInfo = getFullEntryStringUsingManifest(WEB_SERVER_TOMCAT_EMBEDDED_CLASS, WEB_SERVER_TOMCAT_ID);
153         if(webServerInfo != null) {
154             return webServerInfo;
155         }
156         webServerInfo = getFullEntryStringUsingManifest(WEB_SERVER_JETTY_CLASS, WEB_SERVER_JETTY_ID);
157         if(webServerInfo != null) {
158             return webServerInfo;
159         }
160         //WildFly must be before JBoss
161         webServerInfo = getWildFlyEntryInfo();
162         if(webServerInfo != null) {
163             return webServerInfo;
164         }
165         webServerInfo = getFullEntryStringUsingManifest(WEB_SERVER_JBOSS_CLASS, WEB_SERVER_JBOSS_ID);
166         if(webServerInfo != null) {
167             return webServerInfo;
168         }
169         webServerInfo = getWebSphereEntryInfo();
170         if(webServerInfo != null) {
171             return webServerInfo;
172         }
173         webServerInfo = getWebLogicEntryInfo();
174         if(webServerInfo != null) {
175             return webServerInfo;
176         }
177 
178         return null;
179     }

202     private static NameAndVersion getWildFlyEntryInfo() {
203         try {
204             if (Classes.isAvailable(WEB_SERVER_WILDFLY_CLASS)) {
205                 Package wildFlyPkg = Classes.forName(WEB_SERVER_WILDFLY_CLASS).getPackage();
206                 if (wildFlyPkg != null
207                     && Strings.hasText(wildFlyPkg.getImplementationTitle()) && wildFlyPkg.getImplementationTitle().contains("WildFly")) {
208                         return new NameAndVersion(WEB_SERVER_WILDFLY_ID, wildFlyPkg.getImplementationVersion());
209                 }
210             }
211 
212         } catch (RuntimeException e) {
213             throw e;
214         } catch (Exception e){ //NOPMD
215             //there was a problem obtaining the WildFly version
216         }
217         return null;
218     }

234     /**
235      * WARNING: This method must never be invoked unless we already know that the class identified by the parameter {@code fqcn}
236      * really exists in the classpath. For example, we first need to assure that {@code Classes.isAvailable(fqcn))} is <code>TRUE</code>
237      */
238     private static String getVersionInfoInManifest(String fqcn) {
239         String version = null;
240         //get class package
241         Package thePackage = Classes.forName(fqcn).getPackage();
242         // package could be null in some uberjars
243         if (thePackage != null) {
244             //examine the package object
245             version = thePackage.getSpecificationVersion();
246             if (!Strings.hasText(version)) {
247                 version = thePackage.getImplementationVersion();
248             }
249         }
250         if(!Strings.hasText(version)) {
251             version = UNKNOWN_VERSION;
252         }
253         return version;
254     }

256     /**
257      * This method should only be invoked after already knowing that the class identified by {@code WEB_SERVER_WEBSPHERE_CLASS}
258      * really exists in the classpath. For example, it can be checked that {@code Classes.isAvailable(WEB_SERVER_WEBSPHERE_CLASS))}
259      * is {@code TRUE}
260      */
261     private static String getWebSphereVersion() {
262         try {
263             Class<?> versionClass = Class.forName(WEB_SERVER_WEBSPHERE_CLASS);
264             Object versionInfo = versionClass.newInstance();
265             Method method = versionClass.getDeclaredMethod("runReport", String.class, PrintWriter.class);
266             StringWriter stringWriter = new StringWriter();
267             PrintWriter printWriter = new PrintWriter(stringWriter);
268             method.invoke(versionInfo, "", printWriter);
269             String version = stringWriter.toString();
270             // version looks like this, so we need to "substring" it:
271             //
272             //
273             //IBM WebSphere Product Installation Status Report
274             //--------------------------------------------------------------------------------
275             //
276             //Report at date and time August 13, 2014 1:12:06 PM ART
277             //
278             //Installation
279             //--------------------------------------------------------------------------------
280             //Product Directory        C:\Program Files\IBM\WebSphere\AppServer
281             //Version Directory        C:\Program Files\IBM\WebSphere\AppServer\properties\version
282             //DTD Directory            C:\Program Files\IBM\WebSphere\AppServer\properties\version\dtd
283             //Log Directory            C:\Documents and Settings\All Users\Application Data\IBM\Installation Manager\logs
284             //
285             //Product List
286             //--------------------------------------------------------------------------------
287             //BASETRIAL                installed
288             //
289             //Installed Product
290             //--------------------------------------------------------------------------------
291             //Name                  IBM WebSphere Application Server
292             //Version               8.5.5.2
293             //ID                    BASETRIAL
294             //Build Level           cf021414.01
295             //Build Date            4/8/14
296             //Package               com.ibm.websphere.BASETRIAL.v85_8.5.5002.20140408_1947
297             //Architecture          x86 (32 bit)
298             //Installed Features    IBM 32-bit WebSphere SDK for Java
299             //WebSphere Application Server Full Profile
300 
301             version = version.substring(version.indexOf("Installed Product"));
302             version = version.substring(version.indexOf("Version"));
303             version = version.substring(version.indexOf(" "), version.indexOf("\n")).trim();
304             return version;
305 
306         } catch (RuntimeException e) {
307             throw e;
308         } catch (Exception e) { //NOPMD
309             //there was a problem obtaining the WebSphere version
310         }
311         //returning 'unknown' so we can identify in the User-Agent String that we are not properly handling some WebSphere version
312         return UNKNOWN_VERSION;
313     }

332         return UNKNOWN_VERSION;

335     private static Set<NameAndVersion> oktaComponentsFromVersionMetadata() {
336         Set<NameAndVersion> results = new HashSet<>();
337         try {
338             list(ApplicationInfo.class.getClassLoader().getResources(VERSION_FILE_LOCATION)).stream()
339                     .map(ApplicationInfo::loadProps)
340                     .forEach(properties -> results.addAll(entriesFromOktaVersionMetadata(properties)));
341         } catch (IOException e) { //NOPMD
342             // don't fail when gathering info
343             log.warn("Failed to locate okta component version metadata as a resource: {}", VERSION_FILE_LOCATION);
344         }
345         return results;
346     }
347 
348     private static Set<NameAndVersion> entriesFromOktaVersionMetadata(Properties properties) {
349 
350         if (properties == null) {
351             return Collections.emptySet();
352         }
353 
354         return properties.entrySet().stream()
355             .map(entry -> new NameAndVersion((String) entry.getKey(), (String) entry.getValue()))
356             .collect(Collectors.toSet());
357     }
358 
359     private static Properties loadProps(URL resourceUrl) {
360         try {
361             Properties props = new Properties();
362             props.load(resourceUrl.openStream());
363             return props;
364         } catch (IOException e) {
365             // don't fail when gathering info
366             log.warn("Failed to open properties file: '{}', but this file was detected on your classpath", resourceUrl);
367         }
368         return null;
369     }
