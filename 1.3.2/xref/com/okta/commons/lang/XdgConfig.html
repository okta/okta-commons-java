16  package com.okta.commons.lang;
17  
18  import java.io.File;
19  import java.util.ArrayList;
20  import java.util.Arrays;
21  import java.util.List;
22  import java.util.Map;
23  import java.util.Optional;
24  
25  public final class XdgConfig {
26  
27      private static final String OKTA = "okta";
28      private static final String DOT_OKTA = ".okta";
29  
30      private XdgConfig() {}
31  
32      public static File getConfigFile(String relativeConfigFile) {
33          return new File(getConfigDirectory(), relativeConfigFile);
34      }
35  
36      public static File getConfigDirectory() {
37  
38          List<File> configOptions = new ArrayList<>();
39          Map<String, String> envVars = System.getenv();
40  
41          // 1. XDG_CONFIG_HOME or default
42          configOptions.add(new File(envVars.getOrDefault("XDG_CONFIG_HOME", userHome() + "/.config"), OKTA));
43  
44          // 2. check XDG_CONFIG_DIRS
45          Optional.ofNullable(envVars.getOrDefault("XDG_CONFIG_DIRS", "/etc/xdg"))
46              .map(value -> value.split(File.pathSeparator))
47              .ifPresent(values -> Arrays.stream(values).forEachOrdered(value ->
48                  configOptions.add(new File(value, OKTA))
49              ));
50  
51          // 3. check legacy ~/.okta
52          File defaultDirectory = new File(userHome(), DOT_OKTA);
53          configOptions.add(defaultDirectory);
54  
55          return configOptions.stream()
56              .filter(File::exists)
57              .findFirst()
58              .orElse(defaultDirectory);
59      }
60  
61      private static String userHome() {
62          return System.getProperty("user.home");
63      }
64  }
